Init Project:
    - npm init -y

Library:
    - express: simple server
    - mongoose: connector mongodb
    - morgan: http show
    - multer: upload images
    - dotenv: environment variables
    - cross-env: variables
    - cors: send and receive messages from server
    - fs-extra: filesystem that support promisses
    
npm install express mongoose morgan multer dotenv cross-env cors fs-extra

Server update:
    - npm install nodemon -D

Init Backend Server
    - npx nodemon backend/index.js
    - npm run dev

Manjaro:
    - systemctl start mongodb.service
